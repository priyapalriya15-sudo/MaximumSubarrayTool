#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cout << "Enter the number of elements: ";
    cin >> n;

    vector<int> array(n); // vector of size n
    cout << "Enter the elements of the array: ";
    for (int i = 0; i < n; i++)
        cin >> array[i];

    cout << "You entered: ";
    for (int i = 0; i < n; i++)
        cout << array[i] << " ";
    cout << endl;

    return 0;
}
