#include <iostream>
#include <vector>
#include <climits>
using namespace std;

int main() {
    int n, st, end;
    int sum, minsum = INT_MAX;

    cout << "Enter the number of elements: ";
    cin >> n;

    vector<int> array(n); // vector of size n
    cout << "Enter the elements of the array: ";
    for (int i = 0; i < n; i++)
        cin >> array[i];

    for (st = 0; st < n; st++) {
        sum = 0;
        for (end = st; end < n; end++) {
            sum += array[end];
            minsum = min(sum, minsum);
        }
    }

    cout << "Minimum Subarray Sum: " << minsum << endl;
    return 0;
}
